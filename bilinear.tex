\newcommand{\bilinear}[1][]{
\pgfkeys{/bilinear, default, #1}
\tikzmath{
real \startx, \starty;
real \dy, \fy, \dm, \fm, \axsp;
real \axislinet, \bilinet;
integer \mode;
%Conversion all units to the unit pt
\dy = \dy;
\fy = \fy;
\dm = \dm;
\fm = \fm;
\axsp = \axsp;
\axislinet =\axislinet;
\bilinet = \bilinet;
\arrlen = \arrlen;
\arrwid = \arrwid;
%End conversion, All values are now in pt units
\xmax = \dm + \axsp;
\ymax = \fm + \axsp;
\pyd = \dm - \dy;   %Post yield displacement
\pyf = \fm - \fy;   %Post yield force
\lowlevel = min(-\axsp,\fm - 2*\fy - \axsp);
if \showaxes==0 then {\xlab={};\ylab={};} else {};
if \showstiff==0 then {\kpt={};\kst={};} else {};
}
\begin{scope}[x=1pt, y=1pt, xshift = \startx, yshift=\starty, rotate=0];    % Draw everything in pt units

\ifthenelse{\showlinearrows=1}{
\tikzset{linearrow/.style = {postaction={on each segment={mid arrow=black}}}};
}{\tikzset{linearrow/.style = {}};}

\ifthenelse{\showarrows=1}{
\tikzset{arrowbe/.style = {arrows={->[length=2pt, width=2pt]}}};
\tikzset{arrowben/.style = {arrows={<->[length=2pt, width=2pt]}}};
}{\tikzset{arrowbe/.style = {}};}

\ifthenelse{\myp=1}{
\draw [very thin, dashed] (\dy, 0)--node[at start, below=-1]{\dyt}(\dy, \fy);
\draw [very thin, dashed] (0, \fy)--node[at start, left=-2, yshift=0.1cm]{\fyt}(\dy, \fy);
}{}

\ifthenelse{\mmp=1}{
\draw [very thin, dashed] (\dm, 0)--node[at start, below=-1]{\dmt}(\dm, \fm);
\draw [very thin, dashed] (0, \fm)--node[at start, left=-2]{\fmt}(\dm, \fm);
}{}

\ifthenelse{\mode=1}{
\draw [line width=\axislinet, arrowbe] (-\axsp,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width=\axislinet, arrowbe] (0,-\axsp) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width=\bilinet, draw=black, linearrow]
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm);
}{}

\ifthenelse{\mode=2}{
\draw [line width = \axislinet, arrowbe] (-\axsp,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowbe] (0,-\axsp) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow]
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-\dy, -\fy);
}{}


\ifthenelse{\mode=3}{
\draw [line width = \axislinet, arrowbe] (-\axsp,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowbe] (0,\lowlevel) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow]
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy);
}{}

\ifthenelse{\mode=4}{
\draw [line width = \axislinet, arrowben] (\dm- 2*\dy -\pyd -\axsp,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowben] (0,\fm -2*\fy -\pyf -\axsp) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow] 
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy)
-- node[above, midway]{\kst}++(-\pyd, -\pyf);
}{}


\ifthenelse{\mode=5}{
\draw [line width = \axislinet, arrowben] (-\xmax,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowben] (0,-\ymax) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow] 
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy)
-- node[above, midway]{\kst}++(-2*\pyd, -2*\pyf);
}{}

\ifthenelse{\mode=6}{
\draw [line width = \axislinet, arrowben] (-\xmax,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowben] (0,-\ymax) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow] 
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy)
-- node[above, midway]{\kst}++(-2*\pyd, -2*\pyf)
-- node[right, midway]{\kpt}++(\dy, \fy);
}{}

\ifthenelse{\mode=7}{
\draw [line width = \axislinet, arrowben] (-\xmax,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowben] (0,-\ymax) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow] 
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy)
-- node[above, midway]{\kst}++(-2*\pyd, -2*\pyf)
-- node[right, midway]{\kpt}++(2*\dy, 2*\fy);
}{}

\ifthenelse{\mode=8}{
\draw [line width = \axislinet, arrowben] (-\xmax,0) -- node[at end, below=-2pt]{\xlab}(\xmax,0);
\draw [line width = \axislinet, arrowben] (0,-\ymax) -- node[at end, left=-2pt]{\ylab}(0,\ymax);
\path [line width = \bilinet, draw=black, linearrow] 
(0,0)
-- node[right, midway, behind path, fill=white]{\kpt}(\dy,\fy)
-- node[below, midway]{\kst}(\dm,\fm)
-- node[left, midway]{\kpt}++(-2*\dy, -2*\fy)
-- node[above, midway]{\kst}++(-2*\pyd, -2*\pyf)
-- node[right, midway]{\kpt}++(2*\dy, 2*\fy)
-- node[below, midway]{\kst}++(\pyd, \pyf) ;
}{}
\end{scope}
}
