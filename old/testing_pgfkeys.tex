\documentclass[12pt]{article}
\usepackage{tikz}
\usetikzlibrary{math}

\pgfkeys{
 /frame/.is family, /frame,
  default/.style = {length = 10cm},
  length/.estore in = \mylength,
 }
 
\pgfkeys{
 /dofs/.is family, /dofs,
  default/.style = {startX = 0cm, startY = 0cm, endX = 10cm, endY = 5cm},
  startX/.estore in = \startX,
  startY/.estore in = \startY,
  endX/.estore in = \endX,
  endY/.estore in = \endY,
 }
 
 \newcommand{\myline}[1][]{
 \pgfkeys{/frame, default, #1}
 \tikzmath{
 real \mylength, \mylen;
 \mylen = \mylength+1;
 }
 \draw (0,0) -- (\mylen, 0);
 }

 \newcommand{\dofs}[1][]{
 \pgfkeys{/dofs, default, #1}
 \draw (\startX,\startY) -- (\endX,  \endY);
 }
